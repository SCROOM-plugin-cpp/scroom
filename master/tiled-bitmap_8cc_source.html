33 static Scroom::MemoryBlobs::PageProvider::Ptr createProvider(double width, double height, int bpp)
34 {
35  double tileCount = (width*height) / (TILESIZE * TILESIZE);
36  double tileSize = (bpp / 8.0) * TILESIZE * TILESIZE;
37 
38  double guessedTileSizeAfterCompression = tileSize / 100;
39  const int pagesize = 4096;
40  int pagesPerBlock = std::max(int(ceil(guessedTileSizeAfterCompression / 10 / pagesize)),1);
41 
42  int blockSize = pagesPerBlock*pagesize;
43  int blockCount = std::max(int(ceil(tileCount / 10)), 64);
44 
45  printf("Creating a PageProvider providing %d blocks of %d bytes\n", blockCount, blockSize);
46  return Scroom::MemoryBlobs::PageProvider::create(blockCount, blockSize);
47 }

148 void TiledBitmap::initialize()
149 {
150  int width = bitmapWidth;
151  int height = bitmapHeight;
152  unsigned int i = 0;
153  int bpp = 0;
154  LayerOperations::Ptr lo = ls[i];
155  Layer::Ptr prevLayer;
156  LayerOperations::Ptr prevLo;
157  Scroom::MemoryBlobs::PageProvider::Ptr provider = createProvider(width, height, lo->getBpp());
158  do
159  {
160  if(i<ls.size())
161  lo = ls[i];
162 
163  bpp = lo->getBpp();
164 
165  Layer::Ptr layer = Layer::create(shared_from_this<TiledBitmap>(), i, width, height, bpp, provider);
166  layers.push_back(layer);
167  if(prevLayer)
168  {
169  connect(layer, prevLayer, prevLo);
170  }
171 
172  prevLayer = layer;
173  prevLo = lo;
174  width = (width+7)/8; // Round up
175  height = (height+7)/8;
176  i++;
177  } while (std::max(width, height) > TILESIZE/4);
178 }

194 void TiledBitmap::connect(Layer::Ptr const& layer, Layer::Ptr const& prevLayer,
195  LayerOperations::Ptr prevLo)
196 {
197  int horTileCount = prevLayer->getHorTileCount();
198  int verTileCount = prevLayer->getVerTileCount();
199 
200  std::vector<LayerCoordinator::Ptr> coordinators;
201 
202  for(int j=0; j<verTileCount; j++)
203  {
204  int voffset = j%8;
205  if(!voffset)
206  {
207  // New line of target tiles
208  coordinators.clear();
209  CompressedTileLine& tileLine = layer->getTileLine(j/8);
210  for(unsigned int z=0; z<tileLine.size(); z++)
211  {
212  LayerCoordinator::Ptr lc = LayerCoordinator::create(tileLine[z], prevLo);
213  coordinators.push_back(lc);
214  this->coordinators.push_back(lc);
215  }
216  }
217 
218  for(int i=0; i<horTileCount; i++)
219  {
220  int hoffset = i%8;
221  LayerCoordinator::Ptr lc = coordinators[i/8];
222  lc->addSourceTile(hoffset, voffset, prevLayer->getTile(i,j));
223  }
224  }
225 }

266 void TiledBitmap::redraw(ViewInterface::Ptr const& vi, cairo_t* cr, Scroom::Utils::Rectangle<double> const& presentationArea, int zoom)
267 {
268  TiledBitmapViewData::Ptr viewData = this->viewData[vi];
269  auto scaledRequestedPresentationArea = presentationArea;
270 
271  unsigned int layerNr=0;
272  while(zoom<=-3 && layerNr<layers.size()-1)
273  {
274  layerNr++;
275  zoom+=3;
276  scaledRequestedPresentationArea/=8;
277  }
278  Layer::Ptr layer = layers[layerNr];
279  LayerOperations::Ptr layerOperations = ls[std::min(ls.size()-1, (size_t)layerNr)];
280 
281  const Scroom::Utils::Rectangle<int> actualPresentationArea = layer->getRect();
282  const auto validPresentationArea = scaledRequestedPresentationArea.intersection(actualPresentationArea);
283 
284  const int left = scaledRequestedPresentationArea.getLeft();
285  const int top = scaledRequestedPresentationArea.getTop();
286  const int right = scaledRequestedPresentationArea.getRight();
287  const int bottom = scaledRequestedPresentationArea.getBottom();
288 
289  const int imin = std::max(0, left/TILESIZE);
290  const int imax = (right+TILESIZE-1)/TILESIZE;
291  const int jmin = std::max(0, top/TILESIZE);
292  const int jmax = (bottom+TILESIZE-1)/TILESIZE;
293 
294  viewData->setNeededTiles(layer, imin, imax, jmin, jmax, zoom, layerOperations);
295 
296  const double pixelSize = pixelSizeFromZoom(zoom);
297 
298  layerOperations->initializeCairo(cr);
299 
300  const auto clippedRequestedPresentationArea = scaledRequestedPresentationArea
301  .above(validPresentationArea.getBottom())
302  .leftOf(validPresentationArea.getRight());
303 
304  for(int i=imin; i<imax; i++)
305  {
306  for(int j=jmin; j<jmax; j++)
307  {
308  Scroom::Utils::Point<int> tileIndex(i,j);
309 
310  const auto tileArea = TileAreaForIndex(tileIndex);
311  const auto visibleTileArea = tileArea.intersection(clippedRequestedPresentationArea);
312 
313  const auto tileAreaRect = visibleTileArea - tileArea.getTopLeft();
314  const auto viewAreaRect = (visibleTileArea - clippedRequestedPresentationArea.getTopLeft()) * pixelSize;
315 
316  CompressedTile::Ptr tile = layer->getTile(i,j);
317  TileViewState::Ptr tileViewState = tile->getViewState(vi);
318  Scroom::Utils::Stuff cacheResult = tileViewState->getCacheResult();
319  ConstTile::Ptr t = tile->getConstTileAsync();
320 
321  if(t)
322  {
323  cairo_save(cr);
324  layerOperations->draw(cr, t, tileAreaRect, viewAreaRect, zoom, cacheResult);
325  cairo_restore(cr);
326  }
327  else
328  {
329  layerOperations->drawState(cr, tile->getState(), viewAreaRect);
330  }
331 #ifdef DEBUG_TILES
332  drawTile(cr, tile, viewAreaRect);
333 #endif
334  }
335  }
336 }
