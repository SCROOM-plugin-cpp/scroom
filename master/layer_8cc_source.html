170 void DataFetcher::operator()()
171 {
172  // printf("Attempting to fetch bitmap data for tileRow %d...\n", currentRow);
173  QueueJumper::Ptr qj = QueueJumper::create();
174 
175  threadPool->schedule(qj, REDUCE_PRIO, queue);
176 
177  CompressedTileLine& tileLine = layer->getTileLine(currentRow);
178  std::vector<Tile::Ptr> tiles;
179  for(int x = 0; x < horTileCount; x++)
180  {
181  CompressedTile::Ptr ti = tileLine[x];
182  Scroom::Utils::Stuff s = ti->initialize();
183  tiles.push_back(ti->getTileSync());
184  }
185  int lineCount = std::min(TILESIZE, height-currentRow*TILESIZE);
186 
187  sp->fillTiles(currentRow * TILESIZE, lineCount, TILESIZE, 0, tiles);
188 
189  for(int x = 0; x < horTileCount; x++)
190  {
191  tileLine[x]->reportFinished();
192  }
193 
194  currentRow++;
195  if(currentRow<verTileCount)
196  {
197  DataFetcher successor(*this);
198  if(!qj->setWork(successor))
199  threadPool->schedule(successor, DATAFETCH_PRIO, queue);
200  }
201  else
202  sp->done();
203 }
