296  void ProgressInterfaceMultiplexer::updateProgressState()
297  {
298  ProgressStateInterface::State state = ProgressStateInterface::IDLE;
299  double progress = 0.0;
300  int workers = 0;
301 
302  boost::mutex::scoped_lock l(mut);
303  BOOST_FOREACH(const ChildData::Ptr& child, children)
304  {
305  boost::mutex::scoped_lock child_lock(child->mut);
306  switch(child->state)
307  {
308  case ProgressStateInterface::IDLE:
309  break;
310  case ProgressStateInterface::WAITING:
311  if(state==ProgressStateInterface::IDLE || state==ProgressStateInterface::FINISHED)
312  state = ProgressStateInterface::WAITING;
313  progress+=child->progress;
314  workers++;
315  break;
316  case ProgressStateInterface::WORKING:
317  if(state!=ProgressStateInterface::WORKING)
318  state = ProgressStateInterface::WORKING;
319  progress+=child->progress;
320  workers++;
321  break;
322  case ProgressStateInterface::FINISHED:
323  if(state==ProgressStateInterface::IDLE)
324  state=ProgressStateInterface::FINISHED;
325  progress+=1.0;
326  workers++;
327  break;
328  }
329  }
330 
331  parent->setProgress(state, progress/workers);
332 
333  if(state==ProgressStateInterface::FINISHED)
334  {
335  BOOST_FOREACH(const ChildData::Ptr& child, children)
336  {
337  child->clearFinished();
338  }
339  }
340  }
