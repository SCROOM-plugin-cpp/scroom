29 struct Measurement
30 {
31 public:
32  GdkPoint start;
33  GdkPoint end;
34 
35 public:
36  Measurement(int x, int y) { start.x=x; start.y=y; end=start; }
37  Measurement(GdkPoint start) : start(start), end(start) {}
38 
39  bool endsAt(GdkPoint p) { return end.x==p.x && end.y==p.y; }
40 
41  int width() { return abs(end.x-start.x); }
42  int height() { return abs(end.y-start.y); }
43  double length() { return std::sqrt(std::pow(double(width()),2) + std::pow(double(height()),2)); }
44 };
