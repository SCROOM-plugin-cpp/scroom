40 bool PluginManager::doWork()
41 {
42  bool retval = true;
43  char* path = NULL;
44  char* i=NULL;
45  DIR* dir;
46  struct dirent* content;
47  GModule* plugin;
48 
49  gdk_threads_enter();
50  switch(state)
51  {
52  case FINDING_DIRECTORIES:
53  setStatusBarMessage("Locating plugin directories");
54  path = getenv(SCROOM_PLUGIN_DIRS.c_str());
55  dirs.clear();
56  if(!devMode)
57  dirs.push_back(PLUGIN_DIR);
58  if(path!=NULL)
59  {
60  printf("%s=%s\n", SCROOM_PLUGIN_DIRS.c_str(), path);
61  for(i=path; *i!=0; i++)
62  {
63  if(*i==':')
64  {
65  *i=0;
66  dirs.push_back(path);
67  path=i+1;
68  }
69  }
70  dirs.push_back(path);
71  }
72  currentDir = dirs.begin();
73  files.clear();
74  state = SCANNING_DIRECTORIES;
75  break;
76 
77  case SCANNING_DIRECTORIES:
78  setStatusBarMessage("Scanning plugin directories");
79  if(currentDir!=dirs.end())
80  {
81  printf("Scanning directory: %s\n", currentDir->c_str());
82  dir = opendir(currentDir->c_str());
83  if(dir!=NULL)
84  {
85  while( (content=readdir(dir)))
86  {
87  if(content->d_type==DT_REG ||
88  content->d_type==DT_LNK ||
89  content->d_type==DT_UNKNOWN)
90  {
91  files.push_back(g_build_path(G_DIR_SEPARATOR_S, currentDir->c_str(), content->d_name, NULL));
92  }
93  }
94  closedir(dir);
95  }
96  else
97  {
98  printf("Can't open directory...\n");
99  }
100  currentDir++;
101  }
102  else
103  {
104  state = LOADING_FILES;
105  currentFile=files.begin();
106  }
107  break;
108  case LOADING_FILES:
109  setStatusBarMessage("Loading Plugins");
110  if(currentFile!=files.end())
111  {
112  if(!currentFile->compare(currentFile->size()-3, 3, ".so"))
113  {
114  printf("Reading file: %s\n", currentFile->c_str());
115  plugin = g_module_open(currentFile->c_str(), (GModuleFlags)0);
116  if(plugin)
117  {
118  // Need to pass a gpointer to g_module_symbol. If I pass a
119  // PluginFunc, glib 2.16.6/gcc 4.2.4 will complain about
120  // type-punned pointers.
121  gpointer pgpi;
122  if(g_module_symbol(plugin, "getPluginInformation", &pgpi))
123  {
124  PluginFunc gpi = (PluginFunc)pgpi;
125  if(gpi)
126  {
127  PluginInformationInterface::Ptr pi = (*gpi)();
128  if(pi)
129  {
130  if(pi->pluginApiVersion == PLUGIN_API_VERSION)
131  {
132  pluginInformationList.push_back(PluginInformation(plugin, pi));
133  pi->registerCapabilities(shared_from_this<PluginManager>());
134  plugin = NULL;
135  gpi = NULL;
136  pi.reset();
137  }
138  else
139  {
140  printf("Plugin has incorrect API version %d, instead of %d\n",
141  pi->pluginApiVersion, PLUGIN_API_VERSION);
142  }
143  }
144  else
145  {
146  printf("GetPluginInformation returned NULL\n");
147  }
148  }
149  else
150  {
151  printf("Can't find the getPluginInterface function: %s\n", g_module_error());
152  }
153  }
154  else
155  {
156  printf("Can't lookup symbols: %s\n", g_module_error());
157  }
158 
159  if(plugin)
160  {
161  g_module_close(plugin);
162  }
163  }
164  else
165  {
166  printf("Something went wrong: %s\n", g_module_error());
167  }
168  }
169  else
170  {
171  // printf("Skipping file: %s (doesn't end with \".so\")\n", currentFile->c_str());
172  }
173  currentFile++;
174  }
175  else
176  {
177  state = DONE;
178  }
179  break;
180  case DONE:
181  setStatusBarMessage("Done loading plugins");
182  on_done_loading_plugins();
183  retval = false;
184  break;
185  }
186  gdk_threads_leave();
187  return retval;
188 }
