146 void TransparentOverlayViewInfo::redraw(cairo_t* cr, Scroom::Utils::Rectangle<double> const& pa, int zoom)
147 {
148  using Scroom::Bitmap::BitmapSurface;
149 
150  GdkRectangle presentationArea = pa.toGdkRectangle();
151 
152  Scroom::Utils::Rectangle<double> va = (pa - pa.getTopLeft()) * pixelSizeFromZoom(zoom);
153 
154  GdkRectangle viewArea;
155  viewArea.x=0;
156  viewArea.y=0;
157 
158  if(zoom > 0)
159  {
160  const int pixelSize = 1<<zoom;
161  viewArea.width = presentationArea.width*pixelSize;
162  viewArea.height = presentationArea.height*pixelSize;
163  }
164  else
165  {
166  const int pixelSize = 1<<-zoom;
167  viewArea.width = presentationArea.width/pixelSize;
168  viewArea.height = presentationArea.height/pixelSize;
169  }
170 
171  verify(va.toGdkRectangle() == viewArea);
172 
173  BitmapSurface::Ptr s = BitmapSurface::create(viewArea.width, viewArea.height, CAIRO_FORMAT_ARGB32);
174  cairo_surface_t* surface = s->get();
175  cairo_t* cr_sub = cairo_create(surface);
176 
177  int count=0;
178 
179  for(size_t i=0; i<children.size(); i++)
180  {
181  if(gtk_toggle_button_get_active(GTK_TOGGLE_BUTTON(buttons[i])))
182  {
183  PresentationInterface::Ptr& p = children[i];
184 
185  Colormappable::Ptr c = boost::dynamic_pointer_cast<Colormappable>(p);
186  bool hasTransparentBackground = false;
187  if(c && p->isPropertyDefined(TRANSPARENT_BACKGROUND_PROPERTY_NAME))
188  {
189  if(count==0)
190  c->disableTransparentBackground();
191  else
192  {
193  c->setTransparentBackground();
194  hasTransparentBackground = true;
195  }
196  }
197 
198  cairo_save(cr_sub);
199  cairo_set_operator(cr_sub, CAIRO_OPERATOR_CLEAR);
200  cairo_paint(cr_sub);
201  cairo_restore(cr_sub);
202 
203  cairo_save(cr_sub);
204  p->redraw(childViews[p], cr_sub, presentationArea, zoom);
205  cairo_restore(cr_sub);
206 
207  cairo_set_source_surface(cr, surface, 0, 0);
208 
209  if(hasTransparentBackground)
210  cairo_paint(cr);
211  else
212  cairo_paint_with_alpha(cr, 1.0/(count+1));
213 
214  if(!hasTransparentBackground)
215  count++;
216  }
217  }
218 
219  cairo_destroy(cr_sub);
220 }
