183  template<typename T>
184  Observable<T>::~Observable()
185  {
186  // Destroy strong references to any observers
187  //
188  // The one holding the token of the registration is in control
189  // of the lifetime of the registration objects. Hence, as long
190  // as the Token is valid, a reference to the Registration will
191  // exist, which may, in turn, contain a (strong) reference to an
192  // Observer. This is not desirable. As soon as the observable is
193  // deleted (which is now), it should no longer hold any
194  // references to any Observers. Hence, we should manually reset
195  // references to observers here.
196  for(typename Registration::Ptr registration: registrationMap->values())
197  {
198  registration->o.reset();
199  }
200  }
