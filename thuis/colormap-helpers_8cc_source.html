133 void MonochromeColormapHelper::regenerateColormap()
134 {
135  Color w = whitish;
136  Color b = blackish;
137  if(transparentBackground)
138  {
139  if(inverted)
140  b.setAlpha(0);
141  else
142  w.setAlpha(0);
143 
144  }
145 
146  for(int i=0; i<numberOfColors; i++)
147  {
148  colormap->colors[i] = mix(w, b, 1.0*i/(numberOfColors-1));
149  }
150 }
