99 GtkFileFilterInfoPtr filterInfoFromPath(const std::string& filename)
100 {
101  GtkFileFilterInfoPtr filterInfo;
102 
103  std::unique_ptr<GFile,GObjectUnref<GFile> > file(g_file_new_for_path(filename.c_str()));
104  std::unique_ptr<GFileInfo,GObjectUnref<GFileInfo> > fileInfo(g_file_query_info(file.get(), "standard::*", G_FILE_QUERY_INFO_NONE, NULL, NULL));
105  if(fileInfo)
106  {
107  // g_file_info_get_name(fileInfo) doesn't provide path info.
108  // charpFromString might throw.
109  std::unique_ptr<gchar> filenameCopy(charpFromString(filename));
110  std::unique_ptr<gchar> mime_type(charpFromString(g_content_type_get_mime_type(g_file_info_get_content_type (fileInfo.get()))));
111  std::unique_ptr<gchar> display_name(charpFromString(g_file_info_get_display_name(fileInfo.get())));
112 
113  filterInfo.reset(new GtkFileFilterInfo()); // filterInfo->filename uninitialized, so a delete is dangerous
114 
115  filterInfo->filename = filenameCopy.release();
116  filterInfo->mime_type = mime_type.release();
117  filterInfo->display_name = display_name.release();
118  filterInfo->contains =
119  (GtkFileFilterFlags)(GTK_FILE_FILTER_FILENAME | GTK_FILE_FILTER_DISPLAY_NAME | GTK_FILE_FILTER_MIME_TYPE);
120  }
121  else
122  {
123  throw std::invalid_argument("No fileinfo for file "+filename);
124  }
125 
126  return filterInfo;
127 }

180 PresentationInterface::Ptr loadPresentation(GtkFileFilterInfo const& info)
181 {
182  static std::map<boost::filesystem::path,PresentationInterface::WeakPtr> loadedPresentations;
183  // Create canonical path from filename
184  const std::map<OpenPresentationInterface::Ptr, std::string>& openPresentationInterfaces = PluginManager::getInstance()->getOpenPresentationInterfaces();
185 
186 #ifdef HAVE_BOOST_FILESYSTEM_CANONICAL
187  boost::filesystem::path key(canonical(boost::filesystem::path(info.filename)));
188 #else
189  boost::filesystem::path key(absolute(boost::filesystem::path(info.filename)));
190 #endif
191 
192  PresentationInterface::Ptr presentation = loadedPresentations[key].lock();
193 
194  if(!presentation)
195  {
196  for (auto const& cur : openPresentationInterfaces)
197  {
198  std::list<GtkFileFilter*> filters = cur.first->getFilters();
199  if (filterMatchesInfo(info, filters))
200  {
201  presentation = cur.first->open(info.filename);
202 
203  if (presentation)
204  {
205  loadedPresentations[key] = presentation;
206  on_presentation_created(presentation);
207  break;
208  }
209  }
210  }
211  }
212 
213  if(presentation)
214  {
215  return presentation;
216  }
217  else
218  throw std::invalid_argument("Don't know how to load presentation "+std::string(info.filename));
219 }
