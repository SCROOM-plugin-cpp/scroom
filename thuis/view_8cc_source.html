164 void View::redraw(cairo_t* cr)
165 {
166  if(presentation)
167  {
168  GdkRectangle rect;
169  rect.x=x;
170  rect.y=y;
171  if(zoom>=0)
172  {
173  // Zooming in. Smallest step is 1 presentation pixel, which is more than one window-pixel
174  int pixelSize = 1<<zoom;
175  rect.width = (drawingAreaWidth+pixelSize-1)/pixelSize;
176  rect.height = (drawingAreaHeight+pixelSize-1)/pixelSize;
177  }
178  else
179  {
180  // Zooming out. Smallest step is 1 window-pixel, which is more than one presentation-pixel
181  int pixelSize = 1<<(-zoom);
182  rect.width = drawingAreaWidth*pixelSize;
183  rect.height = drawingAreaHeight*pixelSize;
184 
185  // Round to whole pixels
186  rect.x = (rect.x/pixelSize) * pixelSize;
187  rect.y = (rect.y/pixelSize) * pixelSize;
188  }
189 
190  presentation->redraw(shared_from_this<View>(), cr, rect, zoom);
191 
192  if(measurement)
193  {
194  GdkPoint start = presentationPointToWindowPoint(measurement->start);
195  GdkPoint end = presentationPointToWindowPoint(measurement->end);
196  cairo_set_line_width(cr, 1);
197  cairo_set_source_rgb(cr, 0.75, 0, 0); // Dark Red
198  drawCross(cr, start);
199  drawCross(cr, end);
200  cairo_stroke(cr);
201  cairo_set_source_rgb(cr, 1, 0, 0); // Red
202  cairo_move_to(cr, start.x, start.y);
203  cairo_line_to(cr, end.x, end.y);
204  cairo_stroke(cr);
205  }
206  }
207  else
208  {
209  // A logo here would be nice...
210 
211  // char buffer[] = "View says \"Hi\"";
212  //
213  // cairo_move_to(cr, 50, 50);

342 void View::updateZoom()
343 {
344  if(presentation)
345  {
346  int presentationHeight = presentationRect.height();
347  int presentationWidth = presentationRect.width();
348  int minZoom = 0;
349 
350  while(presentationHeight > drawingAreaHeight/2 || presentationWidth > drawingAreaWidth/2)
351  {
352  presentationHeight >>= 1;
353  presentationWidth >>= 1;
354  minZoom--;
355  }
356 
357  gtk_widget_set_sensitive(GTK_WIDGET(zoomBox), true);
358 
359  int zMax = MaxZoom - minZoom;
360  zMax = std::max(zMax, 1+MaxZoom-zoom);
361  zMax = std::min((size_t)zMax, sizeof(zoomfactor)/sizeof(zoomfactor[0]));
362 
363  gtk_list_store_clear(zoomItems);
364  for(int z=0; z<zMax; z++)
365  {
366  GtkTreeIter iter;
367  gtk_list_store_insert_with_values(zoomItems, &iter, z,
368  COLUMN_TEXT, zoomfactor[z],
369  COLUMN_ZOOM, MaxZoom-z,
370  -1);
371 
372  if(zoom == MaxZoom-z)
373  {
374  gtk_combo_box_set_active_iter(zoomBox, &iter);
375  }
376  }
377  }
378  else
379  {
380  gtk_widget_set_sensitive(GTK_WIDGET(zoomBox), false);
381  gtk_list_store_clear(zoomItems);
382  GtkTreeIter iter;
383  gtk_list_store_insert_with_values(zoomItems, &iter, 0,
384  COLUMN_TEXT, zoomfactor[0],
385  COLUMN_ZOOM, MaxZoom,
386  -1);
387 
388  }
389 }

491 {
492  if(event->direction == GDK_SCROLL_UP ||
493  event->direction == GDK_SCROLL_DOWN)
494  {
495  int newZoom = zoom + ((event->direction == GDK_SCROLL_UP)?1:-1);
496  newZoom = std::min(MaxZoom, newZoom);

498  GtkTreeIter iter;

500  valid;
501  valid = gtk_tree_model_iter_next(GTK_TREE_MODEL(zoomItems), &iter))
502  {
503  GValue value= G_VALUE_INIT;
504  gtk_tree_model_get_value(GTK_TREE_MODEL(zoomItems), &iter, COLUMN_ZOOM, &value);
505  int foundZoom = g_value_get_int(&value);
506 
507  if(foundZoom==newZoom)
508  {
509  on_zoombox_changed(newZoom, event->x, event->y);

513  }

626 void View::on_buttonPress(GdkEventButton* event)
627 {

669 void View::on_motion_notify(GdkEventMotion* event)
670 {
671  if((event->state & GDK_BUTTON1_MASK) && modifiermove == GDK_BUTTON1_MASK)
672  {
673  int newx = x;
674  int newy = y;
675 
676  if(zoom>=0)
677  {
678  const int pixelSize=1<<zoom;
679  if(std::abs(event->x-cachedPoint.x)>=pixelSize)
680  {
681  int delta = (int(event->x)-cachedPoint.x)/pixelSize;
682  cachedPoint.x += delta*pixelSize;
683  newx-=delta;
684  }
685  if(std::abs(event->y-cachedPoint.y)>=pixelSize)
686  {
687  int delta = (int(event->y)-cachedPoint.y)/pixelSize;
688  cachedPoint.y += delta*pixelSize;
689  newy-=delta;
690  }
691  }
692  else
693  {
694  const int pixelSize=1<<-zoom;
695  newx-=(event->x-cachedPoint.x)*pixelSize;
696  newy-=(event->y-cachedPoint.y)*pixelSize;
697  cachedPoint = eventToPoint(event);

884  std::map<PresentationInterface::WeakPtr,GtkWidget*>::iterator end = presentations.end();
885

889  next++;
890

893  GtkWidget* m = cur->second;

895  {
896  // Do nothing

912  }
913  else if(!p && m)
914  {
915  // Remove menu item, then remove this element from the map
916  cur->second = NULL;
917  gtk_widget_destroy(m);
918  presentations.erase(cur);
919  }
920  else if(!p && !m)
921  {
922  // Remove this element from the map (menu already gone)
923  presentations.erase(cur);
924  }
925  else
926  {
927  // This cannot happen
928  printf("PANIC! Logic error in view.cc\n");
929  }
931
