50 bool TiffPresentation::load(const std::string& fileName)
51 {
52  try
53  {
54  this->fileName = fileName;
55  tif = TIFFOpen(fileName.c_str(), "r");
56  if (!tif)
57  {
58  // Todo: report error
59  printf("PANIC: Failed to open file %s\n", fileName.c_str());
60  return false;
61  }
62 
63  uint16 spp;
64  if (1 != TIFFGetField(tif, TIFFTAG_SAMPLESPERPIXEL, &spp))
65  spp = 1; // Default value, according to tiff spec
66  if (spp != 1 && spp != 3)
67  {
68  printf("PANIC: Samples per pixel is not 1 or 3, but %d. Giving up\n", spp);
69  return false;
70  }
71  this->spp = spp;
72 
73  TIFFGetFieldChecked(tif, TIFFTAG_IMAGEWIDTH, &width);
74  TIFFGetFieldChecked(tif, TIFFTAG_IMAGELENGTH, &height);
75 
76  uint16 bps;
77  if( 1 != TIFFGetField(tif, TIFFTAG_BITSPERSAMPLE, &bps))
78  {
79  if(spp==1)
80  bps = 1;
81  else
82  bps = 8;
83  }
84  else
85  {
86  if(spp==3)
87  {
88  if(bps!=8)
89  {
90  printf("PANIC: Bits per sample is not 8, but %d. Giving up\n", bps);
91  return false;
92  }
93  }
94  }
95  this->bps = bps;
96 
97  Colormap::Ptr originalColormap;
98 
99  uint16 *r, *g, *b;
100  int result = TIFFGetField(tif, TIFFTAG_COLORMAP, &r, &g, &b);
101  if (result == 1)
102  {
103  originalColormap = Colormap::create();
104  originalColormap->name = "Original";
105  int count = 1 << bps;
106  originalColormap->colors.resize(count);
107 
108  for (int i = 0; i < count; i++)
109  {
110  originalColormap->colors[i] = Color(1.0 * r[i] / 0xFFFF,
111  1.0 * g[i] / 0xFFFF, 1.0 * b[i] / 0xFFFF);
112  }
113 
114  colormapHelper = ColormapHelper::create(originalColormap);
115  }
116 
117  uint16 photometric;
118  TIFFGetFieldChecked(tif, TIFFTAG_PHOTOMETRIC, &photometric);
119  switch (photometric)
120  {
121  case PHOTOMETRIC_MINISBLACK:
122  if (originalColormap)
123  printf("WEIRD: Tiff contains a colormap, but photometric isn't palette\n");
124 
125  if (bps == 1 || bps == 8)
126  colormapHelper = MonochromeColormapHelper::create(2);
127  else
128  colormapHelper = MonochromeColormapHelper::create(1 << bps);
129 
130  properties[MONOCHROME_COLORMAPPABLE_PROPERTY_NAME] = "";
131  break;
132 
133  case PHOTOMETRIC_MINISWHITE:
134  if (originalColormap)
135  printf("WEIRD: Tiff contains a colormap, but photometric isn't palette\n");
136 
137  if (bps == 1 || bps == 8)
138  colormapHelper = MonochromeColormapHelper::createInverted(2);
139  else
140  colormapHelper = MonochromeColormapHelper::createInverted(1 << bps);
141 
142  properties[MONOCHROME_COLORMAPPABLE_PROPERTY_NAME] = "";
143  break;
144 
145  case PHOTOMETRIC_PALETTE:
146  if (!originalColormap)
147  {
148  printf("WEIRD: Photometric is palette, but tiff doesn't contain a colormap\n");
149  colormapHelper = ColormapHelper::create(1 << bps);
150  }
151  break;
152 
153  case PHOTOMETRIC_RGB:
154  if (originalColormap)
155  printf("WEIRD: Tiff contains a colormap, but photometric isn't palette\n");
156  break;
157 
158  default:
159  printf("PANIC: Unrecognized value for photometric\n");
160  return false;
161  }
162 
163  float resolutionX;
164  float resolutionY;
165  uint16 resolutionUnit;
166 
167  if(TIFFGetField(tif, TIFFTAG_XRESOLUTION, &resolutionX) &&
168  TIFFGetField(tif, TIFFTAG_YRESOLUTION, &resolutionY) &&
169  TIFFGetField(tif, TIFFTAG_RESOLUTIONUNIT, &resolutionUnit) &&
170  resolutionUnit == RESUNIT_NONE)
171  {
172  transformationData = TransformationData::create();
173  transformationData->setAspectRatio(1/resolutionX, 1/resolutionY);
174  }
175  else
176  {
177  resolutionX = 1;
178  resolutionY = 1;
179  }
180 
181  printf("This bitmap has size %d*%d, aspect ratio %.1f*%.1f\n",
182  width, height, 1/resolutionX, 1/resolutionY);
183 
184  LayerSpec ls;
185 
186  if (spp == 3)
187  {
188  ls.push_back(Operations24bpp::create());
189  }
190  else if (bps == 2 || bps == 4 || photometric == PHOTOMETRIC_PALETTE)
191  {
192  ls.push_back(
193  Operations::create(colormapHelper, bps));
194  ls.push_back(
195  OperationsColormapped::create(colormapHelper,
196  bps));
197  properties[COLORMAPPABLE_PROPERTY_NAME] = "";
198  }
199  else if (bps == 1)
200  {
201  ls.push_back(
202  Operations1bpp::create(colormapHelper));
203  ls.push_back(
204  Operations8bpp::create(colormapHelper));
205  }
206  else if (bps == 8)
207  {
208  ls.push_back(
209  Operations8bpp::create(colormapHelper));
210  }
211  else
212  {
213  printf("PANIC: %d bits per pixel not supported\n", bps);
214  return false;
215  }
216 
217  tbi = createTiledBitmap(width, height, ls);
218  tbi->setSource(shared_from_this<SourcePresentation>());
219  return true;
220  } catch (const std::exception& ex)
221  {
222  printf("PANIC: %s\n", ex.what());
223  return false;
224  }
225 }

317 void TiffPresentation::fillTiles(int startLine, int lineCount, int tileWidth,
318  int firstTile, std::vector<Tile::Ptr>& tiles)
319 {
320  // printf("Filling lines %d to %d, tile %d to %d (tileWidth = %d)\n",
321  // startLine, startLine+lineCount,
322  // firstTile, (int)(firstTile+tiles.size()),
323  // tileWidth);
324 
325  const tsize_t scanLineSize = TIFFScanlineSize(tif);
326  const int tileStride = tileWidth*spp*bps/8;
327  byte row[scanLineSize];
328 
329  const int tileCount = tiles.size();
330  byte* dataPtr[tileCount];
331  for (int tile = 0; tile < tileCount; tile++)
332  {
333  dataPtr[tile] = tiles[tile]->data.get();
334  }
335 
336  for (int i = 0; i < lineCount; i++)
337  {
338  TIFFReadScanline(tif, (tdata_t) row, startLine + i);
339 
340  for (int tile = 0; tile < tileCount - 1; tile++)
341  {
342  memcpy(dataPtr[tile],
343  row + (firstTile + tile) * tileStride,
344  tileStride);
345  dataPtr[tile] += tileStride;
346  }
347  memcpy(dataPtr[tileCount - 1],
348  row + (firstTile + tileCount - 1) * tileStride,
349  scanLineSize - (firstTile + tileCount - 1) * tileStride);
350  dataPtr[tileCount - 1] += tileStride;
351  }
352 }
