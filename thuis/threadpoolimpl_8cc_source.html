258 void ThreadPool::work(ThreadPool::PrivateData::Ptr priv)
259 {
260  boost::mutex::scoped_lock lock(priv->mut);
261  while(priv->alive)
262  {
263  if(priv->jobcount>0)
264  {
265  priv->jobcount--;
266  lock.unlock();
267  do_one(priv);
268  lock.lock();
269  }
270  else
271  {
272  priv->cond.wait(lock);
273  }
274  }
275 
276  bool busy = priv->completeAllJobsBeforeDestruction;
277  while(busy)
278  {
279  if(priv->jobcount>0)
280  {
281  priv->jobcount--;
282  lock.unlock();
283  do_one(priv);
284  lock.lock();
285  }
286  else
287  {
288  busy = false;
289  }
290  }

293 void ThreadPool::do_one(ThreadPool::PrivateData::Ptr priv)
294 {
295  ThreadPool::Job job;
296 
297  {
298  boost::mutex::scoped_lock lock(priv->mut);
299 
300  while(!priv->jobs.empty() && priv->jobs.begin()->second.empty())
301  priv->jobs.erase(priv->jobs.begin());
302 
303  if(!priv->jobs.empty() && !priv->jobs.begin()->second.empty())
304  {
305  job = priv->jobs.begin()->second.front();
306  priv->jobs.begin()->second.pop();
307  }
308  else
309  {
310  printf("PANIC: JobQueue empty while it shouldn't be\n");
311  }
312  }
313 
314  if(job.queue)
315  {
316  QueueLock l(job.queue);
317  if(l.queueExists())
318  {
319  boost::this_thread::disable_interruption while_executing_jobs;
320  job.fn();
321  }
322  }
