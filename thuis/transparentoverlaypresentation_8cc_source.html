73 void TransparentOverlayPresentation::setOptimalColor(PresentationInterface::Ptr const& p)
74 {
75  Colormappable::Ptr c = boost::dynamic_pointer_cast<Colormappable>(p);
76  if(c)
77  {
78  std::map<Color, int, ColorComparer> currentColors;
79  for(PresentationInterface::Ptr const& child: children)
80  {
81  Colormappable::Ptr cChild = boost::dynamic_pointer_cast<Colormappable>(child);
82  if(cChild && child->isPropertyDefined(MONOCHROME_COLORMAPPABLE_PROPERTY_NAME))
83  {
84  currentColors[cChild->getMonochromeColor()]++;
85  }
86  }
87 
88  Color minimumColor=c->getMonochromeColor();
89  int minimumColorValue = currentColors[minimumColor];
90 
91  for(Color const& color: colors)
92  {
93  if(currentColors[color] < minimumColorValue)
94  {
95  minimumColor = color;
96  minimumColorValue = currentColors[color];
97  }
98  }
99 
100  c->setMonochromeColor(minimumColor);
101  }
102 }
